int anythingToInt(dynamic someVariable) {
  try {
    // Replace everything except 0-9
    final number = someVariable.replaceAll(RegExp(r'[^0-9]'), '');
    int variableNumber = int.parse(number);
    return variableNumber;
  } catch (e) {
    return 0;
  }
}
